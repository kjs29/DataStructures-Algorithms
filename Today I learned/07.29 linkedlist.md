My first implementation of LinkedList

(verified on [Design Linked List 707. LeetCode](https://leetcode.com/problems/design-linked-list/description/))

```py
class Node:
    def __init__(self, data, next=None):
        self.data = data
        self.next = next

class MyLinkedList:
    def __init__(self, data=None):
        if data is not None:
            self.head = Node(data)
            self.size = 1
        else:
            self.head = None
            self.size = 0

    def get(self, index: int) -> int:
        current = self.head
        if index >= self.size or index < 0 or current is None:
            return -1
        current_index = 0
        while True:
            if current_index == index:
                return current.data
            current = current.next
            current_index += 1

    def addAtHead(self, val: int) -> None:
        current = self.head
        new = Node(val)
        if current is None:
            self.head = new
        else:
            new.next = current
            self.head = new
        self.size += 1

    def addAtTail(self, val: int) -> None:
        current = self.head
        new = Node(val)
        if current is None:
            self.head = new
        else:
            while True:
                if current.next is None:
                    current.next = new
                    break
                current = current.next
        self.size += 1

    def addAtIndex(self, index: int, val: int) -> None:
        if index == self.size:
            self.addAtTail(val)
            return
        elif index > self.size:
            return
        elif index == 0:
            current = self.head
            new = Node(val)
            new.next = current
            self.head = new
        else:
            current = self.head
            new = Node(val)
            current_index = 0
            if current is None:
                self.head = new
            else:
                for i in range(index-1):
                    current = current.next
                next = current.next
                new.next = next
                current.next = new
        self.size += 1

    def deleteAtIndex(self, index: int) -> None:
        current = self.head
        if index < 0 or index >= self.size:
            return
        if index == 0:
            self.head = current.next
        elif index == self.size - 1:
            for i in range(index-1):
                current = current.next
            current.next = None
        else:
            current_index = 0
            for i in range(index-1):
                current = current.next
            next_next = current.next.next
            current.next = next_next
        self.size -= 1
    
    def print_list(self):
        current =self.head
        if current is None:
            print("Empty!")
        else:
            while current:
                print(current.data, end=" ")
                current = current.next
        print()
```

Five key operations:

### Access:

`get(self, index)`

### Insert:

`addAtHead(self, value)`

`addAtTail(self, value)`

`addAtIndex(self, value)`

### Delete:

`deleteAtIndex(self, index)`
